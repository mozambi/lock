import android.app.Activity;
import android.os.Bundle;
import android.view.View;
import android.widget.*;

public class MainActivity extends Activity {

    TextView x;
    LinearLayout l;
    RelativeLayout r;
    TableLayout t;

    @Override
    protected void onCreate(Bundle s) {
        super.onCreate(s);
        setContentView(R.layout.activity_main);

        x = (TextView) findViewById(R.id.x);
        l = (LinearLayout) findViewById(R.id.l);
        r = (RelativeLayout) findViewById(R.id.r);
        t = (TableLayout) findViewById(R.id.t);

        findViewById(R.id.bl).setOnClickListener(new View.OnClickListener() {
            public void onClick(View v) { show("LinearLayout", l, r, t); }
        });
        findViewById(R.id.br).setOnClickListener(new View.OnClickListener() {
            public void onClick(View v) { show("RelativeLayout", r, l, t); }
        });
        findViewById(R.id.bt).setOnClickListener(new View.OnClickListener() {
            public void onClick(View v) { show("TableLayout", t, l, r); }
        });
    }

    void show(String s, View v1, View v2, View v3) {
        x.setText("Currently showing: " + s);
        v1.setVisibility(View.VISIBLE);
        v2.setVisibility(View.GONE);
        v3.setVisibility(View.GONE);
    }
}
